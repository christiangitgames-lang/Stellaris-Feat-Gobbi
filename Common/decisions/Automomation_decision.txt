decision_automate_planet_incorporate = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_automate_planet_incorporate

	resources = {
		category = decisions
		cost = {
			minor_artifacts = 100
		}
	}

	potential = {
		owner = { 
			OR = {
				is_hive_empire = yes 
				AND = {
						is_machine_empire = yes
						NOT = {
							has_civic = civic_machine_servitor
						}
					prev = { NOT = { has_designation = col_nanotech_world } }
						
				}
			}
		}
		NOR = { 
			has_modifier = "planet_artifact_relays_machine"
			has_modifier = "planet_artifact_relays_hivemind"
			has_modifier = "auto_decision_automate_planet_incorporate"
		}
	}

	effect = {
		add_modifier = { modifier = "auto_decision_automate_planet_incorporate" }
		hidden_effect = {
			if = {
				limit = { owner = { is_hive_empire = yes } }
				add_modifier = { modifier = "planet_artifact_relays_hivemind" days = 3600 }
			}
			else = { add_modifier = { modifier = "planet_artifact_relays_machine"  days = 3600 } }
		}
	}

	ai_weight = {
		weight = 0
	}
}
