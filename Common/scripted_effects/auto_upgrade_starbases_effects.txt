
auto_upgrade_starbases_check = {
	if = { limit = { has_starbase_size = starbase_outpost }
		auto_upgrade_starbase = {
			TECH_1 = tech_starbase_2
			EVENT = lazy_upg_sta_events.2
			ALLOYS = @starbase_outpost_alloys
			DAYS = 360
			DAYSMESSAGE = 330
		}
	}else_if = { limit = { has_starbase_size = starbase_starport }
		auto_upgrade_starbase = {
			TECH_1 = tech_starbase_3
			EVENT = lazy_upg_sta_events.3
			ALLOYS = @starbase_starport_alloys
			DAYS = 720
			DAYSMESSAGE = 30
		}
	}else_if = { limit = { has_starbase_size = starbase_starhold }
		auto_upgrade_starbase = {
			TECH_1 = tech_starbase_4
			EVENT = lazy_upg_sta_events.4
			ALLOYS = @starbase_starhold_alloys
			DAYS = 1080
			DAYSMESSAGE = 30
		}
	}else_if = { limit = { has_starbase_size = starbase_starfortress }
		auto_upgrade_starbase = {
			TECH_1 = tech_starbase_5
			EVENT = lazy_upg_sta_events.5
			ALLOYS = @starbase_starfortress_alloys
			DAYS = 1440
			DAYSMESSAGE = 30
		}
	}
}

auto_upgrade_starbase = {
	if = {
		limit = {
			prev = {
				[[TECH_1]
					has_technology = $TECH_1$
				]
				[[ALLOYS]
					resource_stockpile_compare = {
						resource = alloys
						value >= $ALLOYS$
						mult = value:positive_modifier_mult|MODIFIER|starbase_upgrade_cost_mult|
					}
				]
			}
		}
		#solar_system = { set_name = "upgrade" }
		starbase_event = {
			id = $EVENT$
			days = $DAYS$
		}
		set_starbase_flag = auto_upgrading_starbase_flag
		
		prev = {
			create_message = { #AUTO_STARBASE_MESSAGE_TYPE
				type = AUTO_STARBASE_MESSAGE_TYPE
				localization = AUTO_STARBASE_MESSAGE_TYPE_DESC
				days = $DAYSMESSAGE$
				target = prev.solar_system
				variable = {
					type = name
					localization = FLEET
					scope = prev
				}
				variable = {
					type = name
					localization = SYSTEM
					scope = prev.solar_system
				}
			}
			[[ALLOYS]
				add_resource = {
					alloys = $ALLOYS$
					mult = value:negative_modifier_mult|MODIFIER|starbase_upgrade_cost_mult|
				}
			]
		}
	}
}
