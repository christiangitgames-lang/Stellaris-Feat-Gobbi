have_one_system_nearly = {
	OR = {
		any_system_within_border = {
			has_hyperlane_to = from
		}
		any_country = {
			any_system_within_border = {
				has_hyperlane_to = from
			}
			OR = {
				root = { is_overlord_to = prev }
				is_in_federation_with = root
			}
		}
		any_country = {
			any_system_within_border = {
				has_hyperlane_to = from
			}
			is_at_war_with = from
			OR = {
				has_defensive_pact = root
				root = { is_guaranteeing = prev }
			}
			any_system_within_border = {
				has_hyperlane_to = root
			}
		}
	}
}
