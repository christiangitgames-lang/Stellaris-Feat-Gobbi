cb_claim = {
	show_in_diplomacy = yes
	show_notification = no
	potential = {
		is_country_type = default
		NOT = { is_in_federation_with = from }
	}
	is_valid = {
		OR = {
			has_claim = FROM
			from = {
				any_subject = {
					root = { has_claim = prev }
				}
			}
		}
		has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
		}
	}
	proxy_war_resources = {
		category = proxy_war
		cost = {
			influence = @proxy_influence_120
		}
	}
	on_proxy_war_start = {
		pay_proxy_war_additional_resources = { VALUE = @pay_proxy_war_additional_resources_30 }
	}
}

cb_subjugation = {
	show_in_diplomacy = no
	show_notification = no
	potential = {
		OR = {
			is_country_type = default
			is_country_type = awakened_fallen_empire
		}
		NOT = { is_in_federation_with = from }
		is_subject = no
	}
	is_valid = {
		NOR = {
			is_overlord_to = FROM
			has_crisis_level = crisis_level_1
		}
		from = { is_subject = no }
		OR = {
			is_country_type = awakened_fallen_empire
			relative_power = {
				who = FROM
				category = all
				value > equivalent
			}
		}
	}
	destroy_if = {
		NOT = { is_country_type = awakened_fallen_empire }
		has_crisis_level = crisis_level_1
		relative_power = {
			who = FROM
			category = all
			value < superior
		}
	}
	proxy_war_resources = {
		category = proxy_war
		cost = {
			influence = @proxy_influence_200
		}
	}
	on_proxy_war_start = {
		pay_proxy_war_additional_resources = { VALUE = @pay_proxy_war_additional_resources_50 }
	}
}

						OR = {
							is_active_resolution = "resolution_mutualdefense_castigation_proclamation"

							OR = {
								has_modifier = resolution_denounced
								is_in_breach_of_any = yes
							}
						}
					}
				}
			}
		}
		#has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
