		id = sfg_utility.1
		hide_window = yes
	
		is_triggered_only = yes
		
		trigger = {
			is_ai = no
			NOT = { has_country_flag = was_human }
		}
		immediate = {
			set_country_flag = was_human
		}
}
country_event = {
	#	cg_ai.1	yearly checked - if multiplayer game see message every 10 anni
	
	id = sfg_utility.2
	hide_window = yes

	is_triggered_only = yes
	
	trigger = {
		has_country_flag = player_host
	}
	immediate = {
		if = {
			limit = {
				OR = {
					is_debug = yes
					any_country = {
						NOT = { has_country_flag = player_host }
						is_ai = no
					}
				}
			}
			random_country = {
				limit = { has_country_flag = player_host }
				country_event = { id = sfg_utility.3 }
			}
		}
	}
}

country_event = {#call  bychange civic update arc furnace produciton.
	id = sfg_utility.4
	hide_window = yes

	is_triggered_only = yes
	trigger = {
		check_variable = {
			which = arc_furnace_counter
			value >= 1
		}
	}
	immediate = {
		every_system_within_border = {
			limit = {
				system_has_arc_furnace = yes
			}
			system_event = { id = sfg_utility.41 }
		}
		
	}
}

system_event = {#aggiorno depositi
	id = sfg_utility.41
	hide_window = yes

	is_triggered_only = yes

		system_has_arc_furnace = yes
	}

					has_modifier = orbital_arc_furnace_4_mod
				}
			}
			inline_script = {
				script = arc_furnace/remove_deposit
			}
			inline_script = {
				script = arc_furnace/add_deposit
			}
			
		}
	}
